בעמוד:
final/pitch-activity.html

מצאנו את התקלה המדויקת:
הכותרת קיימת:
<h1 id="adminTrigger" title="פתיחת ניהול" style="cursor:pointer">בונות את הפיץ׳ שלנו</h1>

והמודאל קיים:
#adminModal

אבל בדיקה בדפדפן הראתה:
document.getElementById("adminTrigger").click();
document.getElementById("adminModal").className;

מחזיר:
"admin-modal no-print"

כלומר לחיצה על adminTrigger לא מוסיפה class open ל־adminModal.
המסקנה: event listener לא מחובר בפועל, או שהפונקציה attachEvents נעצרת לפני החיבור.

יש לבצע תיקון חזק וממוקד:

1. לא להסתמך על attachEvents בלבד לפתיחת האדמין.
יש להוסיף binding עצמאי, מוקדם ופשוט לפתיחת המודאל.

2. להוסיף פונקציה גלובלית/עצמאית ברורה:

function bindAdminTrigger() {
  var trigger = document.getElementById("adminTrigger");
  var modal = document.getElementById("adminModal");

  if (!trigger || !modal) {
    console.warn("[pitch-activity] Admin trigger or modal not found");
    return;
  }

  trigger.onclick = function () {
    modal.classList.add("open");
  };

  trigger.addEventListener("click", function () {
    modal.classList.add("open");
  });
}

3. לקרוא ל־bindAdminTrigger בשני מקומות:
- מיד אחרי שה־DOM מוכן:
document.addEventListener("DOMContentLoaded", bindAdminTrigger);

- וגם בסוף init(), אחרי attachEvents():
bindAdminTrigger();

המטרה: גם אם attachEvents נכשל באמצע בגלל שגיאה אחרת, פתיחת האדמין עדיין תעבוד.

4. להוסיף fallback נוסף עם event delegation:
document.addEventListener("click", function (event) {
  var trigger = event.target.closest("#adminTrigger");
  if (!trigger) return;

  var modal = document.getElementById("adminModal");
  if (modal) modal.classList.add("open");
});
